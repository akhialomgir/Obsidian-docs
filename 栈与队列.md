# 232.用栈实现队列

```js
var MyQueue = function () {
    this.inStack = [];
    this.outStack = [];
};

/** 
 * @param {number} x
 * @return {void}
 */
MyQueue.prototype.push = function (x) { //输入时不需要交换一遍
    this.inStack.push(x);
};

/**
 * @return {number}
 */
MyQueue.prototype.pop = function () {
    if (!this.outStack.length) this.in2out();
    return this.outStack.pop();
};

/**
 * @return {number}
 */
MyQueue.prototype.peek = function () {
    if (!this.outStack.length) this.in2out();
    return this.outStack.at(-1); //重要，最后一个元素的简洁方法
};

/**
 * @return {boolean}
 */
MyQueue.prototype.empty = function () {
    return this.outStack.length === 0 && this.inStack.length === 0;
};

MyQueue.prototype.in2out = function () {
    while (this.inStack.length) { //直接this.stack会报错，因为数组为空也会执行
        this.outStack.push(this.inStack.pop());
    }
}
```

```js
var MyStack = function() {
    this.stack=[];
};

/** 
 * @param {number} x
 * @return {void}
 */
MyStack.prototype.push = function(x) {
    this.stack.push(x);
    for(let i=0;i<this.stack.length-1;i++)this.stack.push(this.stack.shift());
};

/**
 * @return {number}
 */
MyStack.prototype.pop = function() {
    return this.stack.shift();
};

/**
 * @return {number}
 */
MyStack.prototype.top = function() {
    return this.stack[0];
};

/**
 * @return {boolean}
 */
MyStack.prototype.empty = function() {
    return this.stack.length===0;
```

# 150. 逆波兰表达式求值

```js
var evalRPN = function (tokens) {
    const symbols = new Set(["+", "-", "*", "/"]);
    const stack = [];
    for (const c of tokens) {
        if (symbols.has(c)) {
	        // parseInt 转换为整型，截去小数部分
            const val2 = parseInt(stack.pop()), val1 = parseInt(stack.pop());
            if (c === "+") stack.push(val1 + val2);
            else if (c === "-") stack.push(val1 - val2);
            else if (c === "*") stack.push(val1 * val2);
            else stack.push(parseInt(val1 / val2));
        } else stack.push(c);
    }
    return stack[0];
};
```

# 239. 滑动窗口最大值

```js
var maxSlidingWindow = function (nums, k) {
    const ans = [];
    const q = [];//单调队列（双端）
    for (let i = 0; i < nums.length; i++) {
        //入队 保持递减
        while (q.length && nums[q.at(-1)] < nums[i]) q.pop();
        q.push(i);
        //出队 维护滑动窗口长度
        if (i - q[0] >= k) q.shift();
        //记录答案
        if (i >= k - 1) ans.push(nums[q[0]]);
    }
    return ans;
};
```

# 1438. 绝对差不超过限制的最长连续子数组

```js
var longestSubarray = function (nums, limit) {
    let ans = l = 0;
    const qMax = [], qMin = [];
    for (let i = 0; i < nums.length; i++) {
        //入队
        //递减，维护最大值
        while (qMax.length && nums[qMax.at(-1)] <= nums[i]) qMax.pop();
        qMax.push(i);
        //递增，维护最小值
        while (qMin.length && nums[qMin.at(-1)] >= nums[i]) qMin.pop(); 
        qMin.push(i);
        //比较队首的最大最小值，但是插值不符时如何修改？需要改变最大or最小
        //贪心：右移右端点直到不满足条件，再右移左端点
        while (qMax.length && qMin.length && Math.abs(nums[qMax[0]] - nums[qMin[0]]) > limit) {
            if (l === qMax[0]) qMax.shift();
            else if (l === qMin[0]) qMin.shift();
            l++;
        }
        //统计结果
        ans = Math.max(ans, i - l + 1);
    }
    return ans;
};
```

```js
//小根堆实现优先队列
class MinHeap {
    constructor() {
        this.heap = [];
    }
    swap(i1, i2) {
        const temp = this.heap[i1];
        this.heap[i1] = this.heap[i2];
        this.heap[i2] = temp;
    }
    getParentIndex(i) {
        // return Math.floor((i - 1) / 2);
        return (i - 1) >> 1;
    }
    getLeftIndex(i) {
        return i * 2 + 1;
    }
    getRightIndex(i) {
        return i * 2 + 2;
    }
    //自底向上调整
    shiftUp(index) {
	    //递归终止条件
        if (index === 0) {
            return;
        }
        const parentIndex = this.getParentIndex(index);
        //如果父节点大于当前节点 向上必然存在，不用检验
        if (this.heap[parentIndex].value > this.heap[index].value) {
	        //交换当前节点与父节点
            this.swap(parentIndex, index);
            //向上调整
            this.shiftUp(parentIndex);
        }
    }
    //新插入节点首先插入到末尾，然后自底向上递归调整
    insert(value) {
        this.heap.push(value);
        // 递归上移
        this.shiftUp(this.heap.length - 1);
    }
    //自顶向下调整
    shiftDown(index) {
        const leftIndex = this.getLeftIndex(index);
        const rightIndex = this.getRightIndex(index);
        //左节点存在且小于父节点 向下需要检验节点是否为空
        if (this.heap[leftIndex] && this.heap[leftIndex].value < this.heap[index].value) {
            this.swap(leftIndex, index);
            this.shiftDown(leftIndex);
        }
        //右节点存在且小于父节点 向下需要检验节点是否为空
        if (this.heap[rightIndex] && this.heap[rightIndex].value < this.heap[index].value) {
            this.swap(rightIndex, index);
            this.shiftDown(rightIndex);
        }
    }
    //用最后一个节点替换头节点，在自上而下调整
    pop() {
        this.heap[0] = this.heap.pop();
        this.shiftDown(0);
    }
    peek() {
        return this.heap[0];
    }
    size() {
        return this.heap.length;
    }
}
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number[]}
 */
var topKFrequent = function (nums, k) {
    const map = new Map();
    //使用Map对象计数
    nums.forEach(n => {
        map.set(n, map.has(n) ? map.get(n) + 1 : 1);
    });
    const h = new MinHeap();
    //插入节点，如果堆大小大于k则弹出头节点
    map.forEach((value, key) => {
        h.insert({ value, key });
        if (h.size() > k) {
            h.pop();
        }
    });
    return h.heap.map(a => a.key);
};
```
