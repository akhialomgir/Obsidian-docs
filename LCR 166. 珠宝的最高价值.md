```js
//原地修改
//空间复杂度：m*n
var jewelleryValue = function (frame) {
    const m = frame.length, n = frame[0].length;
    for (let i = 0; i < m; i++) {
        for (let j = 0; j < n; j++) {
            frame[i][j] += Math.max(i ? frame[i - 1][j] : 0, j ? frame[i][j - 1] : 0);
        }
    }
    return frame[m - 1][n - 1];
};
```

```js
//空间复杂度：n
var jewelleryValue = function (frame) {
    const m = frame.length, n = frame[0].length;
    const dp = Array(n).fill(0);
    for (let i = 0; i < m; i++) {
        dp[0] += frame[i][0];
        for (let j = 1; j < n; j++) {
            dp[j] = Math.max(dp[j - 1], dp[j]) + frame[i][j];
        }
    }
    return dp[n - 1];
};
```
