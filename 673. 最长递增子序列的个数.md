var findNumberOfLIS = function (nums) {
    const n = nums.length;
    const f = Array(n).fill(1);
    const g = Array(n).fill(1);
    let m = 1;
    for (let i = 0; i < n; i++) {
        for (let j = 0; j < i; j++) {
            if (nums[j] < nums[i]) {
                if (f[i] < f[j] + 1) {
                    f[i] = f[j] + 1;
                    g[i] = g[j];
                } else if (f[i] === f[j] + 1) g[i] += g[j];
            }
        }
        m = Math.max(m, f[i]);
    }
    let ans = 0;
    for (let i = 0; i < n; i++)if (f[i] === m) ans += g[i];
    return ans;
};
