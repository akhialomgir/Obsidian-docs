var longestCommonSubsequence = function (text1, text2) {
    const n = text1.length, m = text2.length;
    const f = Array(m + 1).fill(0);
    for (const x of text1) {
        let prev = f[0]; //使用临时变量存储f[i][j]，防止被覆盖
        for (let j = 0; j < m; j++) {
            const y = text2[j];
            const tmp = f[j + 1];
            //为了避免-1下标，将递推公式+1
            if (x === y) f[j + 1] = prev + 1;
            else f[j + 1] = Math.max(f[j], f[j + 1]);
            //更新prev
            prev = tmp;
        }
    }
    return f[m];
};
