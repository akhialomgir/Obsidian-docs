#algorithm #DP 

```js
//记忆化搜索
var combinationSum4 = function (nums, target) {
    //记忆化 存储dfs[i]的值
    const memo = Array(target + 1);
    function dfs(i) {
        //爬完
        if (!i) return 1;
        //已经计算过 直接读取memo
        if (memo[i] !== undefined) return memo[i];
        //求和
        let res = 0;
        for (const x of nums) {
            if (x <= i) res += dfs(i - x);
        }
        //一步记忆化
        return memo[i] = res;
    }
    return dfs(target);
};
```

```js
//动态规划
var combinationSum4 = function (nums, target) {
    const dp = Array(target + 1).fill(0);
    dp[0] = 1;
    for (let i = 1; i <= target; i++) {
        for (const x of nums) {
            if (x <= i) dp[i] += dp[i - x];
        }
    }
    return dp[target];
};
```
