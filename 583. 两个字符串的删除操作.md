var minDistance = function (word1, word2) {
    const m = word2.length;
    const f = Array.from({ length: m + 1 }).map((v, k) => k);
    for (const x of word1) {
        let prev = f[0];
        f[0]++;
        for (let j = 0; j < m; j++) {
            const y = word2[j];
            const temp = f[j + 1];
            if (x === y) f[j + 1] = prev;
            else f[j + 1] = Math.min(f[j], f[j + 1]) + 1;
            prev = temp;
        }
    }
    return f[m];
};
