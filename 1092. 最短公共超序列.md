var shortestCommonSupersequence = function (str1, str2) {
    const m = str1.length, n = str2.length;
    //此时dp表示最短公共超序列的数量
    const f = Array(m + 1).fill().map(() => Array(n + 1).fill(0));
    f[0] = f[0].map((v, k) => k);
    for (let i = 1; i <= m; i++)f[i][0] = i;
    for (let i = 0; i < m; i++) {
        const x = str1[i];
        for (let j = 0; j < n; j++) {
            const y = str2[j];
            if (x === y) f[i + 1][j + 1] = f[i][j] + 1;
            else f[i + 1][j + 1] = Math.min(f[i][j + 1], f[i + 1][j]) + 1;
        }
    }
    //从末尾出发，到i或j为0结束
    const ans = [];
    let i = m - 1, j = n - 1;
    while (i >= 0 && j >= 0) {
        if (str1[i] === str2[j]) {
            ans.push(str1[i]);
            i--; j--;
        }
        else if (f[i + 1][j + 1] === f[i][j + 1] + 1) {
            ans.push(str1[i]);
            i--;
        } else {
            ans.push(str2[j]);
            j--;
        }
    }
    //加上初始化的一行一列中的开头部分
    return (i >= 0 ? str1.slice(0, i + 1) : "") + (j >= 0 ? str2.slice(0, j + 1) : "") + ans.reverse().join("");
};
