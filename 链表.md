```js
var removeElements = function (head, val) {
    const dummyHead = new ListNode(0, head); //注意需要使用new！
    temp = dummyHead; //只需使用一个变量
    while (temp.next !== null) {
        if (temp.next.val === val) temp.next = temp.next.next; //只更新指针
        else temp = temp.next; //只走一步
    }
    return dummyHead.next;
};
```

# 707.设计链表

```js
var MyLinkedList = function () {
    this.size = 0; //为了方便判定index是否合法
    this.dummy = new ListNode(0); //虚拟头节点
};

/** 
 * @param {number} index
 * @return {number}
 */
MyLinkedList.prototype.get = function (index) {
    if (index < 0 || index >= this.size) return -1;
    let cur = this.dummy;
    for (let i = 0; i <= index; i++) cur = cur.next; //cur需要=
    return cur.val;
};

/** 
 * @param {number} val
 * @return {void}
 */
MyLinkedList.prototype.addAtHead = function (val) {
    this.addAtIndex(0, val);
};

/** 
 * @param {number} val
 * @return {void}
 */
MyLinkedList.prototype.addAtTail = function (val) {
    this.addAtIndex(this.size, val);
};

/** 
 * @param {number} index 
 * @param {number} val
 * @return {void}
 */
MyLinkedList.prototype.addAtIndex = function (index, val) {
    if (index > this.size) return //index===this.size时在尾部插入
    this.size++;
    let prev = this.dummy;
    for (let i = 0; i < index; i++) prev = prev.next; //prev不需要=
    const toAdd = new ListNode(val, prev.next); //命名
    prev.next = toAdd;
};

/** 
 * @param {number} index
 * @return {void}
 */
MyLinkedList.prototype.deleteAtIndex = function (index) {
    if (index < 0 || index >= this.size) return //注意>=
    this.size--;
    let prev = this.dummy;
    for (let i = 0; i < index; i++) prev = prev.next;
    prev.next = prev.next.next;
};
```

# 206.反转链表

```js
//我的解法1：栈（本质递归）
var reverseList = function(head) {
    if(head===null) return head;
    let stack=[];
    let cur=head;
    while(cur.next!==null){
        stack.push(cur);
        cur=cur.next;
    }
    let reversedHead=cur;
    while(stack.length){
        cur.next=stack[stack.length-1];
        cur=stack.pop();
    }
    cur.next=null;
    return reversedHead;
};
```

# 24. 两两交换链表中的节点

```js
//我的解法
var swapPairs = function (head) {
    const ans = head ? head.next ? head.next : head : head; //繁琐的判断
    let prev = new ListNode(0, head);
    while (true) {
        if (!prev.next) return ans; //无用的两步，应该直接写在条件内
        else if (!prev.next.next) return ans; //无用的两步，应该直接写在条件内
        const second = prev.next, first = second.next;
        second.next = first.next;
        first.next = second;
        prev.next = first;
        prev = second;
    }
    return ans;
};
```

# 19.删除链表的倒数第N个节点

```js
//我的解法1：双指针（常数空间）
var removeNthFromEnd = function (head, n) {
    const dummyHead = new ListNode(0, head);
    let left = dummyHead, right = left;
    for (let i = 0; i <= n; i++)right = right.next;
    while (right) {
        left = left.next;
        right = right.next;
    }
    left.next = left.next.next;
    return dummyHead.next;
};
```

```js
//栈（O(n)空间）
var removeNthFromEnd = function (head, n) {
    const dummyHead = new ListNode(0, head);
    let stack = [], cur = dummyHead;
    while (cur) {
        stack.push(cur);
        cur = cur.next;
    }
    for (let i = 0; i < n; i++)stack.pop();
    cur = stack.at(-1);
    cur.next = cur.next.next;
    return dummyHead.next;
}
```

# 面试题 02.07. 链表相交

```js
//我的解法：栈
var getIntersectionNode = function (headA, headB) {
    let stackA = [], stackB = [];
    let A = headA, B = headB;
    while (A) {
        stackA.push(A);
        A = A.next;
    }
    while (B) {
        stackB.push(B);
        B = B.next;
    }
    let ans = null;
    while (stackA.length > 0 && stackB.length > 0 && stackA.at(-1) === stackB.at(-1)) {
        ans = stackA.pop();
        stackB.pop();
    }
    return ans;
};
```

```js
var getIntersectionNode = function (headA, headB) {

        visited.add(temp);
        temp=temp.next;
    }
    temp=headB; //优化：只需要一个变量

```

```js
//最优解：数学+双指针
var getIntersectionNode = function (headA, headB) {
    let A = headA, B = headB;
    while (A !== B) {
        A = A ? A.next : headB; //判断条件为节点本身，所以可以包含a.next===null的情况
        B = B ? B.next : headA;
    }
    return A;
};
```

# 142.环形链表II

```js
//我的解法1：哈希
var detectCycle = function (head) {
    const visited = new Set();
    let cur = head;
    while (cur) {
        if (visited.has(cur)) return cur;
        visited.add(cur);
        cur = cur.next;
    }
    return null;
};
```

```js
        fast = fast.next.next;
        slow = slow.next;
        if (fast === slow) {
            fast = head;
            while (fast !== slow) {
                fast = fast.next;
                slow = slow.next;
            }

        }
    }
    return null;
};
```
