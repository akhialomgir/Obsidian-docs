```js
var numberOfWays = function (n, x) {
    const f = Array(n + 1).fill(0);
    f[0] = 1;
    let s = 0;
    //注意取等
    for (let i = 1; i ** x <= n; i++) {
        const v = i ** x;
        s = Math.min(n, s + v);
        for (let c = s; c >= v; c--)f[c] += f[c - v];
    }
    return f[n] % (1e9 + 7);
};
```
