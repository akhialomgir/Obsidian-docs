```js
var isSymmetric = function (root) {
    let level = [root];

        const nextLevel = [];
        while (level.length) {
            const node = level.shift();
            if (node) {
                nextLevel.push(node.left);

const isPalindrome = function (q) {
    let l = 0, r = q.length - 1;
    while (l < r) {
        if (q[l] !== q[r]) return false;
        l++;
        r--;
    }
    return true;
}
```

```js
var isBalanced = function (root) {
    return recur(root) !== -1; //如果结果中有-1说明有不平衡
};

const recur = function (cur) {
    if (!cur) return 0;
    const leftDepth = recur(cur.left);
    if(leftDepth === -1) return -1; //剪枝 必须再右子树递归前开始
    const rightDepth = recur(cur.right);
    if (rightDepth === -1 || Math.abs(leftDepth - rightDepth) > 1) return -1; //剪枝
    return Math.max(leftDepth, rightDepth) + 1;
}
```

```js
//位运算+二分查找
var countNodes = function(root) {
	//头节点为0
    if (!root) return 0;
	//找到最大层数level
    let level = 0, node = root;
    while (node.left) {
        node = node.left;
        level++;
    }
    //二分查找
    //low为1是因为特定位长的0不好表示且最左已知
    let low = 1 << level, high = (1 << (level + 1)) - 1;
    while (low < high) {
        const mid = Math.floor((high - low + 1) / 2) + low;
        if (exists(root, level, mid)) {
            low = mid;
        } else {
            high = mid - 1;
        }
    }
    return low;
};

//找位数不需要递归，迭代更省空间
const exists = (root, level, k) => {
    let bits = 1 << (level - 1); //转换为二进制数取一位
    let node = root;
    while (node !== null && bits > 0) {
        if (!(bits & k)) {
            node = node.left;
        } else {
            node = node.right;
        }
        bits >>= 1;
    }
    return node;
}
```

```js
//ps:坐标写法好难！跑不出来！
var buildTree = function(inorder, postorder) {
    const n = inorder.length;
    const index = new Map();
    for (let i = 0; i < n; i++) {
        index.set(inorder[i], i);
    }

    function dfs(inL, inR, postL, postR) {
        if (postL === postR) { // 空节点
            return null;
        }
        const leftSize = index.get(postorder[postR - 1]) - inL; // 左子树的大小
        const left = dfs(inL, inL + leftSize, postL, postL + leftSize);
        const right = dfs(inL + leftSize + 1, inR, postL + leftSize, postR - 1);
        return new TreeNode(postorder[postR - 1], left, right);
    }
    return dfs(0, n, 0, n); // 左闭右开区间
};
```

```js
var findMode = function(root) {
    let base = 0, count = 0, maxCount = 0;
    let answer = [];

    const update = (x) => {
        if (x === base) {
            ++count;
        } else {
            count = 1;
            base = x;
        }
        if (count === maxCount) {
            answer.push(base);
        }
        if (count > maxCount) {
            maxCount = count;
            answer = [base];
```
```js
        }

    let cur = root, pre = null;
    while (cur !== null) {
	    //cur.left为空，“左子树不需要访问”，所以访问当前节点并向右（先序遍历）
        if (cur.left === null) {
            update(cur.val); //更新
            cur = cur.right;
            continue;
        }
        //cur.left非空
        pre = cur.left; //pre单次循环内重新设置
        //pre不断向右 排除已经放置的情况
        while (pre.right !== null && pre.right !== cur) {
            pre = pre.right;
        }
        //pre.right为空
        if (pre.right === null) {
            pre.right = cur; //设置索引
            cur = cur.left; //cur左移
        } else {//pre.right===cur 说明“左子树访问完成”
            pre.right = null; //复位索引
            update(cur.val); //访问当前节点
            cur = cur.right;
        }
    }
    return answer;
};
```

```js
    else root.right = insertIntoBST(root.right, val);
    return root;
};
```

# 450. 删除二叉搜索树中的节点

```js
var deleteNode = function (root, key) {
    if (!root) return null;
    if (root.val === key) {
        if (!root.left && !root.right) return null;
        else if (!root.left) return root.right;
        else if (!root.right) return root.left;
        else {
            let successor = root.right;//后继节点
            while (successor.left) successor = successor.left;
            //找到了successor
            root.right = deleteNode(root.right, successor.val); //递归替换
            //在右子树中删除了successor
            successor.left = root.left;
            successor.right = root.right;
            //将successor替换到root上
            return successor;
        }
    }
    if (root.val > key) root.left = deleteNode(root.left, key);
    else root.right = deleteNode(root.right, key);
    return root;
};
```

# 669. 修剪二叉搜索树

```js
var trimBST = function (root, low, high) {
    if (!root) return null;
    if (root.val > high) return trimBST(root.left, low, high);
    if (root.val < low) return trimBST(root.right, low, high);
    root.left = trimBST(root.left, low, high);
    root.right = trimBST(root.right, low, high);
    return root;
};
```

# 108.将有序数组转换为二叉搜索树

```js
var sortedArrayToBST = function (nums) {
    const recur = function (left, right) {
        if (left > right) return null;
        const mid = Math.floor((left + right) / 2);
        const node = new TreeNode(nums[mid]);
        node.left = recur(left, mid - 1);
        node.right = recur(mid + 1, right);
        return node;
    }
    return recur(0, nums.length - 1);
};
```

# 538.把二叉搜索树转换为累加树

```js
var convertBST = function (root) {

        if (!node) return;
        recur(node.right);
        node.val += sum;
        sum = node.val;
        recur(node.left);
```

```js
//Morris遍历
var convertBST = function (root) {
    let sum = 0;
    const update = function (node) {
        node.val += sum;
        sum = node.val;
    }

    let cur = root, pre;
    while (cur) {
        if (!cur.right) {
            update(cur);
            cur = cur.left;
            continue;
        }
        pre = cur.right;
        while (pre.left && pre.left !== cur) pre = pre.left;
        if (!pre.left) {
            pre.left = cur;
            cur = cur.right;
        } else {
            pre.left = null;
            update(cur);
            cur = cur.left;
        }
    }
    return root;
};
```
